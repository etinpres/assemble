def run_hook(tool_name: str, command: str):
    payload = json.dumps({
        "tool_name": tool_name,
        "tool_input": {"command": command},
    })
    proc = subprocess.run(
        ["bash", str(HOOK)],
        input=payload,
        capture_output=True,
        text=True,
        timeout=10,
    )
    return proc

    assert "메인 Claude의 runs/" in proc.stderr or "차단됨" in proc.stderr

           'python3 -c "from server import write_run_artifact; write_run_artifact(\\"r\\", \\"PRD.md\\", \\"x\\")"')

def test_d_python3_no_runs_passes():
