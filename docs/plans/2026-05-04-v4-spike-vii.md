```python
def run_dir_path(run_id: str) -> Path:
    """Return the absolute run directory path for `run_id`. Does not create it.

    Companion to `run_artifact_path` for callers that need just the directory
    (e.g. `{{RUN_DIR}}` token substitution). Reuses the same basename
    validation as `run_artifact_path` to keep the safety contract identical —
    inlined here because the existing `_validate_components` requires a
    second `filename` arg that doesn't apply.
    """
    if not run_id:
        raise ValueError("unsafe run_id: empty")
    if "/" in run_id or "\\" in run_id:
        raise ValueError(f"unsafe run_id: contains separator: {run_id!r}")
    if run_id.startswith("."):
        raise ValueError(f"unsafe run_id: starts with '.': {run_id!r}")
    if run_id != Path(run_id).name:
        raise ValueError(f"unsafe run_id: not a plain basename: {run_id!r}")
    return _runs_dir() / run_id
```

```python
def substitute_inputs(prompt_text: str, inputs: dict) -> str:
    """Substitute `{{KEY}}` placeholders within the `## Inputs` section only.
    ...
    Spike VII Track A: when `RUN_ID` is present and `RUN_DIR` is not,
    `RUN_DIR` is auto-derived as the absolute run dir path. Caller can
    pass `RUN_DIR` explicitly to override (e.g. dogfood / tests).
    """
    if not inputs:
        return prompt_text
    enriched = dict(inputs)
    if "RUN_ID" in enriched and "RUN_DIR" not in enriched:
        # Local import: avoids circular (server.run_dir already imports
        # from server, harness is imported by server/__init__.py).
        from server.run_dir import run_dir_path
        enriched["RUN_DIR"] = str(run_dir_path(str(enriched["RUN_ID"])))
    match = _INPUTS_SECTION_RE.search(prompt_text)
    if match is None:
        return prompt_text
    header, body, next_marker = match.group(1), match.group(2), match.group(3)
    for k, v in enriched.items():
        body = body.replace("{{" + k + "}}", str(v))
    return (
        prompt_text[: match.start()]
        + header + body + next_marker
        + prompt_text[match.end() :]
    )
```

```python
# tests/unit/test_run_dir_token_invariant.py
"""Spike VII Track A regression: bundle prompts must never use
relative `runs/{{RUN_ID}}/...` paths — only absolute `{{RUN_DIR}}/...`.

Sub-agent cwd is unspecified at dispatch time; relative paths resolve
against the SKILL package dir (~/.claude/skills/assemble/), not the
canonical run dir under ~/.claude/channels/assemble/runs/. Spike VI
B-11 real-dispatch only worked due to a manual symlink stop-gap.
"""
from pathlib import Path


SKILL_ROOT = Path(__file__).resolve().parents[2]
BUNDLED = SKILL_ROOT / "bundled"


def test_no_relative_run_path_in_prompts():
    offenders = []
    for prompt in BUNDLED.glob("*/prompts/**/*.md"):
        text = prompt.read_text(encoding="utf-8")
        if "runs/{{RUN_ID}}" in text:
            offenders.append(str(prompt.relative_to(BUNDLED)))
    assert not offenders, (
        f"Prompts must use {{{{RUN_DIR}}}}/... not runs/{{{{RUN_ID}}}}/.... "
        f"Offenders: {offenders}"
    )


def test_run_dir_token_present_in_path_writing_bundles():
    """Bundles that write artifacts must reference {{RUN_DIR}}."""
    must_have = ["reviewer", "builder", "debugger"]
    for bundle in must_have:
        bd = BUNDLED / bundle / "prompts"
        if not bd.exists():
            continue
        all_text = "\n".join(p.read_text() for p in bd.glob("**/*.md"))
        assert "{{RUN_DIR}}" in all_text, (
            f"{bundle}: lost RUN_DIR token after migration"
        )
```

```python
_WROTE_RE = re.compile(r"^WROTE: (.+)$", re.MULTILINE)


def extract_wrote_paths(stdout: str) -> list[str]:
    """Return all `WROTE: <path>` paths from sub-agent stdout, in order.

    Spike VII Track B: anchor at column 0 (MULTILINE `^`) to ignore
    inline `WROTE:` literals in prose. Caller takes `paths[-1]` for
```

## Deviation log (2026-05-04 follow-up commit)

Recorded after ship to keep this plan honest about how reality diverged from the written steps. All deviations were behavior-preserving; ship gate (B-12 12/12 PASS) absorbed them.

### B1 — `_INPUTS_SECTION_RE` regex extension

Plan said the regex was unchanged; implementer extended `(\n## Inputs...)` → `((?:^|\n)## Inputs...)` because `test_save_block_replace_calls_unchanged`'s prompt starts with `## Inputs` at character 0. Backward-compatible: every prior production prompt starts with a title before `## Inputs`, so the new SOF branch is additive. **Follow-up tests added** at `tests/unit/test_substitute_inputs_run_dir.py::test_inputs_at_char_zero_matched` and `::test_inputs_after_preamble_still_matched` to pin both shapes.

### C2 — builder occurrence count

Plan claimed 4 occurrences in builder prompts; actual grep counted 5 (`scope_step2.md` ×1, `test_step3.md` ×2, `verify_step5.md` ×1, plus a save-block-adjacent line that fit the same migration). All 5 migrated correctly; the spec count was off by one because the spec's pre-scan missed an occurrence.

### D1 — `contracts.json` schema

Plan's proposed entry used 3 fields (`id`, `summary`, `test`); the actual project enforces a 5-field schema (`id`, `phrase`, `spec_path`, `section`, `tag`) via `tests/contracts/test_contracts_meta.py`. Implementer added all 5 + the 2 plan extras as a temporary hybrid. **Follow-up cleanup** drops the 2 extras — the regression test the plan wanted to pin (`test_run_dir_token_invariant.py`) is enforced via the test file existing and being collected; the contracts.json `phrase` + `tag` carry the human-readable invariant text. Plan's contracts schema assumption was wrong because the plan was written without reading `contracts.json` first.

### E2 — Output discipline section scope-down

Plan called for adding a 10-line "Output discipline" section to all 25 WROTE-emitting prompts. Re-reviewed before execution: every WROTE prompt already has a top-line "No other prose. Main parses with regex `^WROTE: (.+)$`" instruction; E1's parser fix (`extract_wrote_paths` last-match) addresses F7 at the parser layer; adding 250 lines of duplicate prose to dispatched prompts inflates token cost with marginal benefit. **Scope-down applied**: appended one clarifying sentence (last-match semantic) only to the 6 reviewer prompts where F7 actually surfaced. Other 19 WROTE prompts unchanged.

### F1 — direct edit (no subagent)

Plan listed F1 as a subagent task; implementer (main session) used the `Edit` tool directly because the change was a 1-line text swap with no test impact. Saved one subagent dispatch round; risk was minimal because the line was already cited in the plan with exact old/new strings.
